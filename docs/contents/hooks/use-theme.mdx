---

  `useTheme` is a custom hook that returns a function for retrieving and
  switching themes.
package_name: "@yamada-ui/core"
with_description: true
---

## Import

```ts
import { useTheme } from "@yamada-ui/react"
```

## Usage

```tsx functional=true
const { theme } = useTheme()

return <Text lineClamp={3}>{JSON.stringify(theme)}</Text>
```

:::note
If you want to learn more about the theme, please check [here](/styled-system/theming).
:::

### Switching Themes

```tsx functional=true
const { themeScheme, changeThemeScheme } = useTheme()

return (
  <>
    <Text>The current scheme is "{themeScheme}"</Text>

    <Wrap mt="md" gap="md">
      <For
        each={[
          { changeScheme: "base", children: "Base Theme" },
          {
            colorScheme: "pink",
            changeScheme: "pink",
            children: <>Pink Theme</>,
          },
          {
            colorScheme: "purple",
            changeScheme: "purple",
            children: <>Purple Theme</>,
          },
          {
            colorScheme: "green",
            changeScheme: "green",
            children: <>Green Theme</>,
          },
        ]}
      >
        {({ colorScheme, changeScheme, children }, index) => (
          <Button
            key={index}
            colorScheme={colorScheme}
            onClick={() => changeThemeScheme(changeScheme)}
          >
            {children}
          </Button>
        )}
      </For>
    </Wrap>
  </>
)
```

:::note status=warning
In order to switch themes, you need to prepare multiple themes. For more details, please check [here](/styled-system/theming/switch-themes).
:::
